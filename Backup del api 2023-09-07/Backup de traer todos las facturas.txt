   public List<Facturas> ObtenerTodasFactura()
        {
            using (var context = new TS_TecnocuisineEntities())
            {
                try
                {
                    var detail = context.Facturas
                        .Where(x => x.id > 0).Include(x => x.DetalleFactura).Include(x => x.TipoDocumento).Include(x => x.Proveedores)
                        .ToList();

                    return detail;

                }
                catch (Exception ex)
                {
                    return new List<Facturas>();
                }
            }
        }
